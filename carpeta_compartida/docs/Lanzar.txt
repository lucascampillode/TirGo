Este documento explica como lanzar comandos a Tiago y como conectarse al Sim de tiago:


LANZAR A TIAGO REAL
source /opt/ros/noetic/setup.bash
source ~/carpeta_compartida/gallium/setup.bash
source ~/carpeta_compartida/setup_env.sh
Ejemplo: rostopic list => ver los topics disponibles

LANZAR A TIAGO SIM
source /opt/ros/noetic/setup.bash
source ~/carpeta_compartida/setup_env_sim.sh
source ~/carpeta_compartida/ros_ws/devel/setup.bash
roscore
launch_tiago_simulation
Ejemplo: rostopic list => ver los topics disponibles


Checks rápidos

¿Estoy en local?

echo $ROS_MASTER_URI   # http://localhost:11311
echo ${ROS_IP:-<vacio>}
pgrep -a rosmaster     # debería haber uno

¿Estoy en remoto?
echo $ROS_MASTER_URI   # http://tiago-222c:11311
echo $ROS_IP           # 10.172.209.165
ping -c1 tiago-222c

LANZAR WEB



export MONGO_URI="mongodb://tirgo_app:tirgo@127.0.0.1:27017/tirgo?authSource=tirgo"
export FLASK_SECRET_KEY="dev-secret"
export TIRGO_HOTWORD="hola"
export TIRGO_STT_TOPIC="/stt/text"
export TIRGO_DEV="1"                      # activa /simulate_hola
export PORT="9001"



cd ~/carpeta_compartida/ros_ws
catkin_make
source devel/setup.bash


rosrun tirgo_ui tirgo_web_server
