Este documento explica como lanzar comandos a Tiago y como conectarse al Sim de tiago:


LANZAR A TIAGO REAL
source /opt/ros/noetic/setup.bash
source ~/carpeta_compartida/gallium/setup.bash
source ~/carpeta_compartida/setup_env.sh
Ejemplo: rostopic list => ver los topics disponibles

LANZAR A TIAGO SIM
source /opt/ros/noetic/setup.bash
source ~/carpeta_compartida/setup_env_sim.sh
source ~/carpeta_compartida/ros_ws/devel/setup.bash
roscore
launch_tiago_simulation
Ejemplo: rostopic list => ver los topics disponibles


rosrun tirgo_mission_server tirgo_mission_server.py
roslaunch stt_vosk stt_vosk.launch
roslaunch tirgo_ui web.launch
rosrun tirgo_mission_server tiago_speech_node.py 

LANZAR ACTIONS

cd ~/carpeta_compartida/ros_ws

source /opt/ros/noetic/setup.bash          # ROS base
source ~/carpeta_compartida/gallium/setup.bash  # stack de PAL
source ~/carpeta_compartida/setup_env.sh   # MASTER_URI, ROS_IP del TIAGo
source devel/setup.bash                    # tu workspace (tirgo_ui, tirgo_msgs, tirgo_mission_server...)



# 1) Arranque
roscore
rosrun tirgo_mission_server tirgo_mission_server.py
roslaunch tirgo_ui web.launch
# Abres la web, haces un pedido â†’ comienza GOING_TO_DISPENSER

# 2) Llegada al dispensador
rostopic pub /tirgo/tiago/arrived std_msgs/Bool "data: true"

# 3) Dispensador listo
rostopic pub /tirgo/dispense/ready std_msgs/Bool "data: true"

# 4) Robot ha cogido el envase
rostopic pub /tirgo/tiago/picked std_msgs/Bool "data: true"

# 5) Robot ha llegado al paciente
rostopic pub /tirgo/tiago/at_patient std_msgs/Bool "data: true"

# 6) Robot ha entregado el envase
rostopic pub /tirgo/tiago/delivered std_msgs/Bool "data: true"

# 7) Robot ha terminado su despedida
rostopic pub /tirgo/tiago/farewell_done std_msgs/Bool "data: true"
